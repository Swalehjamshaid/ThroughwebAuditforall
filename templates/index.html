Quick Start — Pro (CLI)

python web_audit_pro.py \ 
  --url https://example.com \ 
  --max-pages 100 \ 
  --timeout 10 \ 
  --user-agent "WebAuditPro/1.0" \ 
  --respect-robots \ 
  --concurrency 20 \ 
  --pagespeed-api-key YOUR_PSI_KEY \ 
  --gsc-credentials path/to/creds.json \ 
  --gsc-property https://example.com/ \ 
  --semrush-key YOUR_SEMRUSH_KEY \ 
  --ahrefs-key YOUR_AHREFS_KEY \ 
  --out audit_report.json \ 
  --pdf-out audit_report.pdf
